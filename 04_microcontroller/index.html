// define pin D0 for A-1A (PWM Speed) const int A1B = D1; // define pin D1 for A-1B (direction) // setting PWM properties const int freq = 5000; const int resolution = 8; void setup() { pinMode(A1A, OUTPUT); // specify these pins as outputs pinMode(A1B, OUTPUT); ledcAttach(A1A, freq, resolution); ledcWrite(A1A, 0); // start with the motors off digitalWrite(A1B, LOW); } void loop() { // start the motor at a low speed ledcWrite(A1A, 200); digitalWrite(A1B, LOW); delay(3000); // allow the motor to run for 3 seconds // reverse the motor direction at the same speed digitalWrite(A1B, HIGH); // switch direction ledcWrite(A1A, 50); // here the logic for speed is inverted delay(4000); // allow the motor to run for 4 seconds // stop the motor ledcWrite(A1A, 0); // setting both pins LOW stops the motor digitalWrite(A1B, LOW); delay(1000); // keep the motor off for 1 second // start the motor at a low speed ledcWrite(A1A, 200); digitalWrite(A1B, LOW); delay(2000); // allow the motor to run for 2 seconds // reverse the motor direction at the same speed digitalWrite(A1B, HIGH); // switch direction ledcWrite(A1A, 50); // here the logic for speed is inverted delay(2000); // allow the motor to run for 2 seconds // stop the motor ledcWrite(A1A, 0); // setting both pins LOW stops the motor digitalWrite(A1B, LOW); delay(1000); // keep the motor off for 1 second // start the motor at a low speed for a longer time period ledcWrite(A1A, 200); digitalWrite(A1B, LOW); delay(5000); // allow the motor to run for 5 seconds // stop the motor ledcWrite(A1A, 0); // setting both pins LOW stops the motor delay(1000); // keep the motor off for 1 second // reverse the motor direction at the same speed for a longer time period digitalWrite(A1B, HIGH); // switch direction ledcWrite(A1A, 50); // here the logic for speed is inverted delay(5000); // allow the motor to run for 10 seconds // spin back and forth // start the motor at a low speed ledcWrite(A1A, 200); // here the logic for speed is inverted delay(1000); // allow the motor to run for 2 seconds // stop the motor (END) ledcWrite(A1A, 0); // setting both pins LOW stops the motor digitalWrite(A1B, LOW); delay(5000); // keep the motor off for 5 seconds } </code> </pre> <p> Circuit Components: </p> <p> This is the L9110 motor driver.
